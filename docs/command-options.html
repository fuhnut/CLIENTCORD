command options

options let users pass inputs into your commands. clientcord provides typed option constructors for every discord option type.

string options

commands/greet.py
from clientcord.declare import command, options
from clientcord.commands.options import string_option

@options(
    name=string_option(description="who to greet", required=True)
)
@command(name="greet", description="greet someone")
class GreetCommand:
    async def run(self, ctx, name: str):
        await ctx.write(f"hello {name}!")

integer options

commands/roll.py
from clientcord.declare import command, options
from clientcord.commands.options import integer_option

@options(
    sides=integer_option(description="number of sides", required=False)
)
@command(name="roll", description="roll a die")
class RollCommand:
    async def run(self, ctx, sides: int = 6):
        import random
        result = random.randint(1, sides)
        await ctx.write(f"rolled a {result}")

user options

commands/info.py
from clientcord.declare import command, options
from clientcord.commands.options import user_option
from clientcord.models import User

@options(
    target=user_option(description="user to look up", required=True)
)
@command(name="info", description="get user info")
class InfoCommand:
    async def run(self, ctx, target: User):
        await ctx.write(f"user id: {target.id}")
