creating a bot - step by step

step 1: make a config file named cc_init.py so clientcord can pick up your token and intents.

cc_init.py 
from clientcord.intents import Intents

token = "your bot token goes here"
intents = Intents.default()
prefix = "!"

step 2: make your main script file named bot.py. imports must be highly modular as shown. you must import individual files instead of blanket imports.

bot.py 
from clientcord.client import Client
from clientcord.declare import command, event

client = Client()

@command(name="ping", description="ping command")
class Ping:
    async def run(self, ctx):
        await ctx.write("pong")

@event("MESSAGE_CREATE")
async def on_message(message):
    print("message received")

client.load_component(Ping)
client.load_component(on_message)

if __name__ == "__main__":
    client.run()

making commands

step 3: use options decorators to pass user inputs tightly into your parameters.

commands context 
from clientcord.declare import command, options
import clientcord.commands.options as opt
from clientcord.models import User

@options(
    user=opt.user(description="target user", required=True),
    reason=opt.string(description="reason for ban", required=False)
)
@command(name="ban")
class BanCommand:
    async def run(self, ctx, user: User, reason: str = None):
        await ctx.write(f"banned {user.id} for {reason}")

using middlewares

step 4: define logic block filters to stop abusive requests before they reach your command logic.

middlewares rules 
from clientcord.declare import middleware
from clientcord.declare import command

@middleware("is_developer")
async def is_dev(ctx, next_mw, stop):
    if ctx.user.id != "my_id":
        return stop("unauthorized")
    await next_mw()

@command(name="shutdown")
@is_dev
class Shutdown:
    async def run(self, ctx):
        await ctx.write("shutting down")
